function renderSection(section, title, tasks) { const total = tasks.length; const done = tasks.filter(t => isCompleted(t, viewDate)).length; const isCollapsed = sectionsState[section]; // true = свернуто, false = развернуто const box = document.createElement("div"); box.className = "sectionBox"; box.innerHTML = `

${title}

${done}/${total} выполнено
`; const content = box.querySelector(".sectionContent"); tasks.forEach(t => content.appendChild(renderTask(t))); const toggleBtn = box.querySelector(".toggle"); toggleBtn.onclick = () => { sectionsState[section] = !sectionsState[section]; saveSectionsState(); render(); }; return box; }
